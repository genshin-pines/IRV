## 二、读取并显示摄像头数据参考代码（OpenCV）

```python
import cv2


def main():
    rtsp_url = "rtsp://10.126.59.120:8554/live/live1"

    # 使用 FFMPEG 后端并启 TCP 传输, 减少花屏/延迟
    cap = cv2.VideoCapture(rtsp_url, cv2.CAP_FFMPEG)
    cap.set(cv2.CAP_PROP_BUFFERSIZE, 1)

    if not cap.isOpened():
        print(f"无法打开视频流: {rtsp_url}")
        return

    print(f"成功连接到视频流: {rtsp_url}")
    print("按 q 退出，按 s 截图保存为 snapshot.jpg")

    while True:
        ret, frame = cap.read()
        if not ret:
            print("读取帧失败，流可能已断开")
            break

        cv2.imshow("RTSP Stream", frame)

        key = cv2.waitKey(1) & 0xFF
        if key == ord("q"):
            break
        elif key == ord("s"):
            cv2.imwrite("snapshot.jpg", frame)
            print("已保存截图：snapshot.jpg")

    cap.release()
    cv2.destroyAllWindows()


if __name__ == "__main__":
    main()
```

## 三、识别并显示识别后的画面参考代码（以 YOLO 为例）

```python
import cv2
from ultralytics import YOLO


def main():
    rtsp_url = "rtsp://10.126.59.120:8554/live/live1"

    # 加载 YOLOv8 预训练模型（可替换成自己的模型）
    model = YOLO("yolov8n.pt")

    # 读取 RTSP 流
    cap = cv2.VideoCapture(rtsp_url, cv2.CAP_FFMPEG)
    cap.set(cv2.CAP_PROP_BUFFERSIZE, 1)

    if not cap.isOpened():
        print(f"无法打开视频流: {rtsp_url}")
        return

    print(f"成功连接到视频流: {rtsp_url}")
    print("按 q 退出")

    while True:
        ret, frame = cap.read()
        if not ret:
            print("读取帧失败，流可能已断开")
            break

        # YOLO 推理（stream=True 要省内存）
        results = model(frame, verbose=False)

        # 在图上绘制识别框和标签
        annotated = results[0].plot()

        cv2.imshow("YOLO Detection", annotated)

        if cv2.waitKey(1) & 0xFF == ord("q"):
            break

    cap.release()
    cv2.destroyAllWindows()


if __name__ == "__main__":
    main()
```
